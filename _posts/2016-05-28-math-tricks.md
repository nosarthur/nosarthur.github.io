$$ \sigma'(x) = \sigma(x)(1-\sigma(x)) $$

$$ 1-\sigma(x) = \sigma(-x) $$

$$ \log\sigma(x) = -\zeta(-x) $$

$$ \zeta'(x) = \sigma(x) $$

$$ \forall x\in(0,1),  \sigma^{-1}(x) = \log\frac{x}{1-x} $$

$$ \forall x>0,  \zeta^{-1}(x) = \log(e^x - 1) $$

$$ \zeta(x)-\zeta(-x) = x $$
